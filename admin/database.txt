CREATE DATABASE ncmea


CREATE TABLE Events (
    Event_ID INT PRIMARY KEY AUTO_INCREMENT,
    Event_Title VARCHAR(225) NOT NULL,
    Event_Content VARCHAR(225) NOT NULL,
    Event_Location VARCHAR(225) NOT NULL,
    Event_Date DATE NOT NULL,
    Event_Image TEXT NOT NULL
)

CREATE TABLE Blog_Cats (
    Blog_Cat_ID INT PRIMARY KEY AUTO_INCREMENT,
    Blog_Cat_Name VARCHAR(225) NOT NULL
)

CREATE TABLE Blogs (
    Blog_ID INT PRIMARY KEY AUTO_INCREMENT,
    Blog_Cat_ID INT NOT NULL,
    Blog_Title VARCHAR(225),
    Blog_Content TEXT NOT NULL,
    Blog_Image_S TEXT,
    Blog_Image_B TEXT,
    FOREIGN KEY(Blog_Cat_ID) REFERENCES Blog_Cats(Blog_Cat_ID) ON UPDATE CASCADE ON DELETE RESTRICT
)

CREATE TABLE Message (
    Message_ID PRIMARY KEY AUTO_INCREMENT,
    Name VARCHAR(225) NOT NULL,
    Phone VARCHAR(14) NOT NULL,
    Email VARCHAR(225) NOT NULL,
    Subject VARCHAR(225) NOT NULL,
    Content VARCHAR(225) NOT NULL
)

CREATE TABLE Users (
    User_ID INT PRIMARY KEY AUTO_INCREMENT,
    Name VARCHAR(225) NOT NULL,
    Email VARCHAR(225) NOT NULL,
    Address VARCHAR(225) NOT NULL, 
    Phone VARCHAR(14) NOT NULL,
    Password VARCHAR(225) NOT NULL,
    Fpassword_Token VARCHAR(225) NOT NULL,
    Session_ID VARCHAR(225),
    Email_enc VARCHAR(225),
    Reg_Date DATE NOT NULL
)
